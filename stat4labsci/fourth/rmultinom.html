# Simulate from a multinomial distribution
rmultinom <-
function(n=100, p=c(0.25,0.5,0.25))
{
  p <- p/sum(p) # ensure that the probabilities sum to 1
  if(any(p < 0)) stop("probabilities p must be non-negative")
  k <- length(p)

  x <- table(factor(sample(1:k, n, repl=TRUE, prob=p), levels=1:k))
  names(x) <- NULL
  x
}
